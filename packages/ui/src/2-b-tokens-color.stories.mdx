import {
  Meta,
  Subtitle,
  ColorPalette,
  ColorItem,
} from "@storybook/addon-docs/blocks";
import { palette } from "./theme/tokens";

<Meta title="Foundations/Design Tokens/Color" />

export const createPalette = (colors) =>
  Object.keys(colors).map((colorKey) => ({
    title: colorKey,
    subtitle: `theme.palette.${colorKey}`,
    colors: Object.values(colors[colorKey]),
  }));

# Color Tokens

<Subtitle>
  Colors are designed to communicate actions, state and direction within the
  interface. We provide light and dark color modes. However, the shape of the
  colors object is same for each mode. This allows us to build components with a
  consistent API.
</Subtitle>

### Palette

<ColorPalette>
  {createPalette(palette).map((colorItem) => (
    <ColorItem key={colorItem.title} {...colorItem} />
  ))}
</ColorPalette>
