import {
  Meta,
  ColorPalette,
  ColorItem,
  Story,
  Preview,
} from "@storybook/addon-docs/blocks";
import { useTheme } from "styled-components";

<Meta title="Foundations/Design Tokens" />

export const Palette = () => {
  const theme = useTheme();
  const palette = Object.keys(theme.colors.palette).map((colorKey) => ({
    title: colorKey,
    subtitle: `theme.colors.palette.${colorKey}`,
    colors: Object.values(theme.colors.palette[colorKey]),
  }));
  return (
    <ColorPalette>
      {palette.map((colorItem) => (
        <ColorItem key={colorItem.title} {...colorItem} />
      ))}
    </ColorPalette>
  );
};

# Color Tokens

> Colors are designed to communicate actions, state and direction within the
> interface. We provide light and dark color modes. However, the shape of the
> colors object is same for each mode. This allows us to build components with a
> consistent API.
